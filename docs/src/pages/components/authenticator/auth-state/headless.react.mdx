You can use [`useAuthenticator`](#access--modify-auth-state) and [Slots](#headers--footers) to provide your own implementation for each screens. For example, you can replace the default `signIn` screen.

```tsx
import { Authenticator, useAuthenticator } from '@aws-amplify/ui-react';

const MySignIn = () => {
  const { isPending, submitForm, updateForm } = useAuthenticator();

  const handleChange = (event: React.FormEvent<HTMLFormElement>) => {
    const { name, value } = event.target;
    updateForm({ name, value });
  };

  const handleSubmit = (event: React.FormEvent<HTMLFormElement>) => {
    event.preventDefault();
    submitForm();
  };

  return (
    <div>
      <h3>Custom Sign In Implementation</h3>

      <form onChange={handleChange} onSubmit={handleSubmit}>
        <fieldset disabled={isPending}>
          <label for="username">Username</label>
          <input
            name="username"
            id="username"
            placeholder="Enter your username"
            autocomplete="username"
          />
          <label for="password">Username</label>
          <input
            name="password"
            id="password"
            type="password"
            placeholder="Enter your username"
            autocomplete="current-password"
          />
        </fieldset>
      </form>
    </div>
  );
};

const components = {
  SignIn: MySignIn,
};

export default function App() {
  return (
    <Authenticator components={components}>
      {({ signOut }) => <button onClick={signOut}>Sign out</button>}
    </Authenticator>
  );
}
```
